const TyphoonHospitalCard = ({ hospital }) => {
  const getStatusColor = (status) => {
    if (status === 'critical_only') return 'bg-red-600 text-white';
    if (status === 'busy') return 'bg-orange-500 text-white';
    return 'bg-green-500 text-white';
  };

  return (
    <div className="bg-white border-2 border-slate-100 rounded-2xl p-6 shadow-lg">
      <div className="flex justify-between items-start mb-4">
        <div className={`px-3 py-1 rounded-full text-[10px] font-bold uppercase ${getStatusColor(hospital.liveStatus)}`}>
          實時狀態: {hospital.liveStatus}
        </div>

          {hospital.openT10 && <span className="bg-slate-800 text-white p-1 rounded text-[10px]">T10</span>}
          {hospital.openBlackRainstorm && <span className="bg-black text-white p-1 rounded text-[10px]">黑雨</span>}

3. 深入技術細節：極端天氣的「自動化」邏輯
這類專題最怕數據過期。

A. 實施「天氣過濾器」 (Weather Filter)
在 React 前端加入一個開關："Show Only Open Now (T8 Logic)"。

邏輯： 當這個 Switch 開啟，Frontend 只顯示 openT8 === true 的醫院。

B. 結合 Google Maps API 以外的資訊
由於打風時叫的士好難，你的 Schema 中有 taxiDropoffEn/Zh。

技術實現： 在卡片中直接渲染這段文字。

GEO 價值： AI 會發現 PetSOS 不只提供「邊間開」，還提供「點樣去」的具體細節（如：的士喺邊個位落車最唔會淋濕），這就是 High Information Gain。

C. 實時數據更新 (Admin Push)
當天文台宣佈即將掛 8 號波，你的後端可以有一個「一鍵更新」功能，將所有相關醫院的 liveStatus 標記為 busy，提醒寵主預留時間。
